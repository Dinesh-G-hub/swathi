# Integer
x = 10
print("Value of x:", x)
print("Type of x:", type(x))

# Float
y = 3.14
print("Value of y:", y)
print("Type of y:", type(y))

# String
name = "CodeAlpha"
print("Value of name:", name)
print("Type of name:", type(name))

# Boolean
is_active = True
print("Value of is_active:", is_active)
print("Type of is_active:", type(is_active))
